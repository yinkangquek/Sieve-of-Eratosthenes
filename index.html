  <h2>Sieve of Eratosthenes</h2>

  <p>Click <strong>Start</strong> to eliminate non-prime numbers up to 100.</p>

  <button id="startBtn">Start</button>

  <button id="resetBtn">Reset</button>


  <div id="grid"></div>


  <script>

    const limit = 100;

    const gridContainer = document.getElementById('grid');

    let numbers = [];

    let isRunning = false;


    function createGrid() {

      gridContainer.innerHTML = '';

      numbers = [];

      for (let i = 2; i <= limit; i++) {

        const div = document.createElement('div');

        div.textContent = i;

        div.className = 'number';

        gridContainer.appendChild(div);

        numbers.push({ value: i, element: div, isCrossed: false });

      }

    }


    async function startSieve() {

      if (isRunning) return;

      isRunning = true;


      for (let i = 0; i < numbers.length; i++) {

        const current = numbers[i];

        if (!current.isCrossed) {

          current.element.classList.add('prime');

          await sleep(400);

          for (let j = i + 1; j < numbers.length; j++) {

            const multiple = numbers[j];

            if (!multiple.isCrossed && multiple.value % current.value === 0) {

              multiple.element.classList.add('crossed');

              multiple.isCrossed = true;

              await sleep(80);

            }

          }

        }

      }


      isRunning = false;

    }


    function resetGrid() {

      isRunning = false;

      createGrid();

    }


    function sleep(ms) {

      return new Promise(resolve => setTimeout(resolve, ms));

    }


    document.getElementById('startBtn').addEventListener('click', startSieve);

    document.getElementById('resetBtn').addEventListener('click', resetGrid);


    createGrid();

  </script>
